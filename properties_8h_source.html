244  json toJson() const {
245  json json_obj = json::object();
246  for (const auto& property : properties_)

248  return json_obj;

257  void fromJson(const json& json_obj) {
258  CHECK(json_obj.is_object());
259 
260  bool at_least_one_value = false;
261  for (const auto& property : properties_) {
262  auto json_it = json_obj.find(property->name());
263  if (json_it != json_obj.end()) {
264  property->setValue(json_it.value());
265  at_least_one_value = true;
266  } else
267  property->setValue(property->defaultValue());
268  }
269 
270  // sanity check, if none of the fields match we have something
271  // completly different than we expected
272  CHECK(json_obj.empty() || properties_.empty() || at_least_one_value);
273  }
274 
275  protected:
276  template <class T>
277  T registerProperty(const char* name,
278  T default_value,
279  T* value,
280  const char* description) {
281  assert(name != nullptr);
282  assert(value != nullptr);
283  properties_.push_back(
284  make_unique<TypedProperty<T>>(this, name, default_value, value, description));
285  return default_value;
286  }
287 
288  private:
289  vector<unique_ptr<Property>> properties_;
290  bool sealed_ = false;
