187 template <class TAG>
188 class PropertySetVariant : public core::NonCopyable {
189  public:
190  using TagType = TAG;
191 
192  public:
193  virtual ~PropertySetVariant() = default;
194 
196  TAG tag() const { return tag_; }
197 
199  void selectCase(TAG tag) {
200  if (cases_.find(tag_) == cases_.end())
201  throw core::Exception("Invalid variant case");
202  tag_ = tag;
203  }
204 
206  const PropertySet* activeCase() const {
207  auto it = cases_.find(tag_);
208  CHECK(it != cases_.end());
209  return it->second;
210  }
211 
213  PropertySet* activeCase() {
214  auto it = cases_.find(tag_);
215  CHECK(it != cases_.end());
216  return it->second;
217  }
218 
223  json toJson() const {
224  json json_obj = json::object();
225  json_obj["tag"] = core::toString(tag_);
226  for (const auto& [case_tag, case_value] : cases_) {
227  json_obj[core::toString(case_tag)] = case_value->toJson();
228  }
229  return json_obj;
230  }
231 
238  void fromJson(const json& json_obj) {
239  CHECK(json_obj.is_object());
240 
241  bool at_least_one_case = false;
242  tag_ = core::fromString<TAG>(json_obj.at("tag"));
243  for (auto& [case_tag, case_value] : cases_) {
244  auto json_it = json_obj.find(core::toString(case_tag));
245  if (json_it != json_obj.end()) {
246  case_value->fromJson(json_it.value());
247  at_least_one_case = true;
248  } else {
249  case_value->resetToDefaultValues();
250  }
251  }
252 
253  // sanity check
254  CHECK(cases_.empty() || at_least_one_case);
255  }
256 
258  void copyFrom(const PropertySetVariant& src) {
259  CHECK(typeid(*this) == typeid(src));
260  CHECK(cases_.size() == src.cases_.size());
261  tag_ = src.tag_;
262  for (const auto& [case_tag, case_value] : src.cases_) {
263  cases_.at(case_tag)->copyFrom(*case_value);
264  }
265  }
266 
267  protected:
268  bool registerCase(TAG tag, PropertySet* property_set) {
269  CHECK(cases_.insert({ tag, property_set }).second, "Duplicate cases");
270  tag_ = tag;
271  return true;
272  }
273 
274  private:
275  TAG tag_ = TAG(-1);
276  map<TAG, PropertySet*> cases_;
277 };

411  CHECK(typeid(*this) == typeid(src));

420  CHECK(!sealed_);

426  if (sealed_)

428  }
429 
435  json toJson() const {
436  json json_obj = json::object();
437  for (const auto& property : properties_) {
438  json_obj[property->name()] = property->toJson();
439  }
440  return json_obj;
441  }
442 
449  void fromJson(const json& json_obj) {
450  CHECK(json_obj.is_object());
451 
452  bool at_least_one_value = false;
453  for (const auto& property : properties_) {
454  auto json_it = json_obj.find(property->name());
455  if (json_it != json_obj.end()) {
456  property->fromJson(json_it.value());
457  at_least_one_value = true;
458  } else
459  property->setValue(property->defaultValue());
460  }
461 
462  // sanity check, if none of the fields match we have something
463  // completely different than we expected
464  CHECK(json_obj.empty() || properties_.empty() || at_least_one_value);
465  }
466 
467  protected:
468  template <class T>
469  T registerProperty(const char* name,
470  T default_value,
471  T* value,
472  const char* description) {
473  assert(name != nullptr);
474  assert(value != nullptr);
475  properties_.push_back(
476  make_unique<TypedProperty<T>>(this, name, default_value, value, description));
477  return default_value;
478  }

480  template <class V, class TAG>
481  bool registerVariant(const char* name,
482  TAG default_case,
483  V* variant,
484  const char* description) {
485  assert(name != nullptr);
486  assert(variant != nullptr);
487  variant->selectCase(default_case);
488  properties_.push_back(
489  make_unique<VariantProperty<V>>(this, name, default_case, variant, description));
490  return true;
491  }
492 
493  private:
494  vector<unique_ptr<Property>> properties_;
495  bool sealed_ = false;
496 };
