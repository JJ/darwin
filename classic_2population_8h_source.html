48  void createPrimordialGeneration(int population_size) override {

52 
53  generation_ = 0;
54  ranked_ = false;
55 
56  genotypes_.resize(population_size);

63  void createNextGeneration() override {
64  CHECK(ranked_);
65 
66  darwin::StageScope stage("Create next generation");
67 
68  ++generation_;
69 
70  vector<GENOTYPE> next_generation(genotypes_.size());
71 
72  atomic<size_t> elite_count = 0;
73  atomic<size_t> babies_count = 0;
74  atomic<size_t> mutate_count = 0;
75 
76  pp::for_each(next_generation, [&](int index, GENOTYPE& genotype) {
77  std::random_device rd;
78  std::default_random_engine rnd(rd());
79  std::uniform_int_distribution<int> dist_parent;
80  std::uniform_real_distribution<double> dist_survive(0, 1);
81  std::bernoulli_distribution dist_mutate_elite(g_config.elite_mutation_chance);
82 
83  auto old_genotype = genotypes_[index];
84  const double old_age = g_config.old_age;
85  const double time_left = old_age > 0 ? (old_age - old_genotype.age) / old_age : 0;
86 
87  const bool viable = old_genotype.age < g_config.larva_age ||
88  old_genotype.fitness >= g_config.min_viable_fitness;
89 
90  // keep the elite population
91  const int elite_limit = max(2, int(genotypes_.size() * g_config.elite_percentage));
92  if (index < elite_limit && old_genotype.fitness > g_config.elite_min_fitness) {
93  // direct reproduction
94  genotype = old_genotype;
95  if (dist_mutate_elite(rnd)) {
96  genotype.mutate();
97  genotype.genealogy = darwin::Genealogy("em", { index });
98  } else {
99  genotype.genealogy = darwin::Genealogy("e", { index });
100  }
101  ++genotype.age;
102  ++elite_count;
103  } else if (index >= 2 && (!viable || dist_survive(rnd) > time_left)) {
104  // pick two parents and produce the offspring
105  int parent1 = dist_parent(rnd) % (index / 2);
106  int parent2 = dist_parent(rnd) % (index / 2);
107 
108  const auto& g1 = genotypes_[parent1];
109  const auto& g2 = genotypes_[parent2];
110 
111  float f1 = fmaxf(g1.fitness, 0);
112  float f2 = fmaxf(g2.fitness, 0);
113 
114  float preference = f1 / (f1 + f2);
115  if (isnan(preference))
116  preference = 0.5f;
117 
118  genotype.inherit(g1, g2, preference);
119  genotype.genealogy = darwin::Genealogy("c", { parent1, parent2 });
120  genotype.mutate();
121  ++babies_count;
122  } else {
123  // last resort, mutate the old genotype
124  genotype = old_genotype;
125  genotype.genealogy = darwin::Genealogy("m", { index });
126  genotype.mutate();
127  ++genotype.age;
128  ++mutate_count;
129  }
130  });
131 
132  std::swap(genotypes_, next_generation);
133 
134  const double population_size = genotypes_.size();
135 
136  log("Next gen stats: %.2f%% elite, %.2f%% babies, %.2f%% mutate\n",
137  (elite_count / population_size) * 100,
138  (babies_count / population_size) * 100,
139  (mutate_count / population_size) * 100);
140 
141  ranked_ = false;
142  }
143 
144  void rankGenotypes() override {
145  CHECK(!ranked_);
146 
147  // sort results by fitness (descending order)
148  std::sort(genotypes_.begin(),
149  genotypes_.end(),
150  [](const GENOTYPE& a, const GENOTYPE& b) { return a.fitness > b.fitness; });
151 
152  // log best fitness values
153  core::log("Fitness values: ");
154  const size_t sample_size = min(size_t(16), genotypes_.size());
155  for (size_t i = 0; i < sample_size; ++i) {
156  log(" %.3f", genotypes_[i].fitness);
157  }
158  log(" ...\n");
159 
160  ranked_ = true;
161  }
162 
163  private:
164  vector<GENOTYPE> genotypes_;
165  int generation_ = 0;
166  bool ranked_ = false;
167 };

169 template <class GENOTYPE>

171  unique_ptr<darwin::Population> create(const core::PropertySet& config,
172  const darwin::Domain& domain) override {
173  g_config.copyFrom(config);
174  g_inputs = domain.inputs();
175  g_outputs = domain.outputs();
176  CHECK(g_inputs > 0);
177  CHECK(g_outputs > 0);
178  ann::setActivationFunction(g_config.activation_function);
179  ann::setGateActivationFunction(g_config.gate_activation_function);
180  return make_unique<Population<GENOTYPE>>();

183  unique_ptr<core::PropertySet> defaultConfig(

188  config->hidden_layers = {};

191  break;
192 
193  case darwin::ComplexityHint::Balanced:
194  case darwin::ComplexityHint::Extra:
195  break;
196  }
