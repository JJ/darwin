117  void setRotation(Scalar angle) {

119  const Scalar s = sin(angle);
120  setIdentity();

128  HMatrix2d operator*(const HMatrix2d& other) {
129  HMatrix2d result;

131  for (int j = 0; j < 3; ++j)
132  for (int k = 0; k < 3; ++k)
133  result.m[i][j] += m[i][k] * other.m[k][j];
